    ```typescript
    type ResolveItemPriceInput = {
      companyId: string;
      itemId: string;
      customerId?: string;
      vendorId?: string;
      transactionDate: string | Date;
    };

    type ResolveItemPriceResult = {
      unitPrice: Prisma.Decimal;
      pricePerKg: Prisma.Decimal | null;
      cylinderWeightKg: Prisma.Decimal | null;
      usingKgPricing: boolean;
    };

    export async function resolveItemPrice(
      tx: Prisma.TransactionClient,
      input: ResolveItemPriceInput,
    ): Promise<ResolveItemPriceResult>
    ```

    Implementation per PR-01-AC2 and PR-01-AC5:
    - Query the applicable ItemPrice: prefer customer-specific (`customerId = input.customerId`) over default (`customerId = null`), and join Item to get `cylinderWeightKg`.
    - Use `findFirst` with:
      ```
      where: {
        itemId: input.itemId,
        customerId: input.customerId ?? null,
        validFrom: { lte: new Date(input.transactionDate) },
        OR: [
          { validTo: null },
          { validTo: { gte: new Date(input.transactionDate) } },
        ],
      },
      orderBy: [{ customerId: "desc" }, { validFrom: "desc" }],
      include: { item: { select: { cylinderWeightKg: true } } },
      ```
    - If `pricePerKg` is set AND `item.cylinderWeightKg` is set: calculate `derivedUnitPrice = pricePerKg × cylinderWeightKg`. Return `{ unitPrice: derivedUnitPrice, pricePerKg, cylinderWeightKg, usingKgPricing: true }`.
    - If `pricePerKg` is NOT set OR `cylinderWeightKg` is null: fall back to fixed price. Return `{ unitPrice: itemPrice.price, pricePerKg: null, cylinderWeightKg: itemPrice.item.cylinderWeightKg, usingKgPricing: false }`.
    - If no ItemPrice found: throw `Error("No price found for item ${itemId}")`.
    - Use `Prisma.Decimal` for all arithmetic (not raw JS numbers).
    - Round the derived unit price to 2 decimal places using `.toDecimalPlaces(2)` to match the existing `Decimal(14,2)` precision on ItemPrice.price (per PR-01-AC2).
